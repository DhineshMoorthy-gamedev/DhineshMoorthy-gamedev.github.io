Config API - Code.gs 
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("RemoteConfig");
  const rows = sheet.getDataRange().getValues();

  let result = {};

  for (let i = 1; i < rows.length; i++) {
    const key = rows[i][0];
    const value = rows[i][1];
    if (key) result[key] = value;
  }

  return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
}

Unity Remote Config Manager

RemoteConfigManager.cs 
using System;
using System.Threading.Tasks;
using UnityEngine;
using UnityEngine.Networking;

public class RemoteConfigManager : MonoBehaviour
{
  public static string remoteConfigURL = "YOUR_CONFIG_URL_HERE";
  
  public string enemySpeed = "";
  public string spawnRate = "";
  public string eventEnabled = "";
  public string status = "Press Load Config";
  public bool loading = false;
  public bool requestLoad = false;

  private void Update()
  {
      if (requestLoad)
      {
          requestLoad = false;
          _ = LoadConfigAsync();
      }
  }

  private async Task LoadConfigAsync()
  {
      loading = true;
      status = "Loading...";

      using UnityWebRequest req = UnityWebRequest.Get(remoteConfigURL);
      var op = req.SendWebRequest();
      
      while (!op.isDone)
          await Task.Yield();

      if (req.result == UnityWebRequest.Result.Success)
      {
          try
          {
              var cfg = JsonUtility.FromJson(req.downloadHandler.text);
              enemySpeed = cfg.enemy_speed;
              spawnRate = cfg.spawn_rate;
              eventEnabled = cfg.event_enabled;
              status = "Config Loaded";
          }
          catch
          {
              status = "JSON Parse Error";
          }
      }
      else
      {
          status = "Error: " + req.error;
      }

      loading = false;
  }
}

[Serializable]
public class ConfigData
{
  public string enemy_speed;
  public string spawn_rate;
  public string event_enabled;
}

Apps Script for Player Management

Add this complete backend logic to your Apps Script:

Player API - Code.gs 
const SHEET_NAME = "Players";

function doPost(e) {
  Logger.log("RAW REQUEST: " + e.postData.contents);

  const data = JSON.parse(e.postData.contents);

  switch (data.action) {
    case "register": return register(data);
    case "login": return login(data);
    case "admin_ban": return adminBan(data);
    case "admin_unban": return adminUnban(data);
    case "admin_getAll": return getAllPlayers();
    default:
      return output({ success: false, message: "Invalid action" });
  }
}

function register(data) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME);
  const values = sheet.getDataRange().getValues();

  const username = data.username;
  const password = data.password;
  const country = data.country || "Unknown";
  const deviceModel = data.deviceModel || "Unknown";

  // Check if user exists
  for (let i = 1; i < values.length; i++) {
    if (values[i][0] === username) {
      return output({ success: false, message: "User already exists" });
    }
  }

  // Add new row
  sheet.appendRow([
    username,
    password,
    "NO",                // banned?
    "",                  // last login
    country,
    deviceModel,
    new Date()           // created at
  ]);

  return output({ success: true, message: "Registered successfully" });
}

function login(data) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME);
  const values = sheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i++) {
    let row = values[i];

    if (row[0] === data.username) {
      // Check ban
      if (row[2] === "YES") {
        return output({ success: false, message: "Player banned" });
      }

      if (row[1].toString() === data.password) {
        // Update last login
        sheet.getRange(i + 1, 4).setValue(new Date());

        // Update country & device
        if (data.country)
          sheet.getRange(i + 1, 5).setValue(data.country);
        if (data.deviceModel)
          sheet.getRange(i + 1, 6).setValue(data.deviceModel);

        return output({
          success: true,
          message: "Login success",
          info: {
            username: row[0],
            country: row[4],
            deviceModel: row[5],
            banned: row[2]
          }
        });
      }

      return output({ success: false, message: "Wrong password" });
    }
  }

  return output({ success: false, message: "User not found" });
}

function adminBan(data) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME);
  const values = sheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i++) {
    if (values[i][0] === data.username) {
      sheet.getRange(i + 1, 3).setValue("YES");
      return output({ success: true, message: "Banned" });
    }
  }
  return output({ success: false, message: "User not found" });
}

function adminUnban(data) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME);
  const values = sheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i++) {
    if (values[i][0] === data.username) {
      sheet.getRange(i + 1, 3).setValue("NO");
      return output({ success: true, message: "Unbanned" });
    }
  }
  return output({ success: false, message: "User not found" });
}

function getAllPlayers() {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME);
  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 7).getValues();

  let players = data.map(r => ({
    username: r[0],
    banned: r[2],
    lastLogin: r[3],
    country: r[4],
    deviceModel: r[5],
    createdAt: r[6]
  }));

  return output({ success: true, players });
}

function output(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

Unity Player Manager Implementation

Now create the Unity script that handles all player operations:

PlayerManager.cs 
using System;
using System.Text;
using System.Threading.Tasks;
using UnityEngine;
using UnityEngine.Networking;

public class PlayerManager : MonoBehaviour
{
    public static string playerDataURL = "YOUR_PLAYER_URL_HERE";

    // Input fields
    public string regUser = "";
    public string regPass = "";
    public string regCountry = "IN";
    public string regStatus = "Ready";
    
    public string logUser = "";
    public string logPass = "";
    public string logStatus = "Ready";
    
    public string adminUser = "";
    public string adminStatus = "Ready";
    
    public AdminPlayerList cachedList;

    private static async Task<string> Post(object payload)
    {
        string json = JsonUtility.ToJson(payload);

        using (UnityWebRequest req = new UnityWebRequest(playerDataURL, "POST"))
        {
            byte[] body = Encoding.UTF8.GetBytes(json);
            req.uploadHandler = new UploadHandlerRaw(body);
            req.downloadHandler = new DownloadHandlerBuffer();

            req.SetRequestHeader("Content-Type", "application/json");
            req.SetRequestHeader("Accept", "application/json");
            req.SetRequestHeader("Access-Control-Allow-Origin", "*");

            var op = req.SendWebRequest();
            while (!op.isDone)
                await Task.Yield();

            return req.downloadHandler.text;
        }
    }

    // Response structures
    [Serializable]
    public class LoginInfo
    {
        public string username;
        public string country;
        public string deviceModel;
        public string banned;
    }

    [Serializable]
    public class LoginResponse
    {
        public bool success;
        public string message;
        public LoginInfo info;
    }

    [Serializable]
    public class RegisterResponse
    {
        public bool success;
        public string message;
    }

    [Serializable]
    public class RegisterData
    {
        public string action;
        public string username;
        public string password;
        public string country;
        public string deviceModel;
    }

    [Serializable]
    public class LoginRequest
    {
        public string action;
        public string username;
        public string password;
        public string country;
        public string deviceModel;
    }

    [Serializable]
    public class AdminPlayer
    {
        public string username;
        public string banned;
        public string lastLogin;
        public string country;
        public string deviceModel;
        public string createdAt;
    }

    [Serializable]
    public class AdminPlayerList
    {
        public bool success;
        public AdminPlayer[] players;
    }

    // API Methods
    public static async Task<RegisterResponse> Register(string username, string password, string country = "Unknown")
    {
        RegisterData d = new RegisterData
        {
            action = "register",
            username = username,
            password = password,
            country = country,
            deviceModel = SystemInfo.deviceModel
        };

        string result = await Post(d);
        return JsonUtility.FromJson<RegisterResponse>(result);
    }

    public static async Task<LoginResponse> Login(string username, string password, string country = "Unknown")
    {
        LoginRequest req = new LoginRequest
        {
            action = "login",
            username = username.Trim(),
            password = password.Trim(),
            country = country,
            deviceModel = SystemInfo.deviceModel
        };

        string result = await Post(req);
        return JsonUtility.FromJson<LoginResponse>(result);
    }

    public static async Task<bool> AdminBan(string username)
    {
        var req = new { action = "admin_ban", username = username };
        string result = await Post(req);
        var r = JsonUtility.FromJson<RegisterResponse>(result);
        return r.success;
    }

    public static async Task<bool> AdminUnban(string username)
    {
        var req = new { action = "admin_unban", username = username };
        string result = await Post(req);
        var r = JsonUtility.FromJson<RegisterResponse>(result);
        return r.success;
    }

    public static async Task<AdminPlayerList> AdminGetAllPlayers()
    {
        var req = new { action = "admin_getAll" };
        string result = await Post(req);
        return JsonUtility.FromJson<AdminPlayerList>(result);
    }

    // UI Methods
    public async void CallRegister()
    {
        if (string.IsNullOrEmpty(regUser) || string.IsNullOrEmpty(regPass))
        {
            regStatus = "Error: Username and Password required";
            return;
        }

        regStatus = "Registering...";

        try
        {
            var res = await Register(regUser, regPass, regCountry);
            regStatus = res.success ? "‚úì " + res.message : "‚úó " + res.message;
            if (res.success)
            {
                regUser = "";
                regPass = "";
            }
        }
        catch (Exception e)
        {
            regStatus = "‚úó Error: " + e.Message;
        }
    }

    public async void CallLogin()
    {
        if (string.IsNullOrEmpty(logUser) || string.IsNullOrEmpty(logPass))
        {
            logStatus = "Error: Username and Password required";
            return;
        }

        logStatus = "Logging in...";

        try
        {
            var res = await Login(logUser, logPass, regCountry);
            
            if (res.success)
            {
                logStatus = "‚úì " + res.message;
                if (res.info != null)
                {
                    logStatus += $"\nCountry: {res.info.country} | Device: {res.info.deviceModel}";
                    if (res.info.banned == "true")
                        logStatus += "\n‚ö† Account is BANNED";
                }
                logPass = "";
            }
            else
            {
                logStatus = "‚úó " + res.message;
            }
        }
        catch (Exception e)
        {
            logStatus = "‚úó Error: " + e.Message;
        }
    }

    public async void CallAdminBan()
    {
        if (string.IsNullOrEmpty(adminUser))
        {
            adminStatus = "Error: Username required";
            return;
        }

        adminStatus = $"Banning {adminUser}...";

        try
        {
            var res = await AdminBan(adminUser);
            adminStatus = res ? $"‚úì Banned {adminUser}" : $"‚úó Failed to ban {adminUser}";
            if (res) await RefreshPlayerList();
        }
        catch (Exception e)
        {
            adminStatus = "‚úó Error: " + e.Message;
        }
    }

    public async void CallAdminUnban()
    {
        if (string.IsNullOrEmpty(adminUser))
        {
            adminStatus = "Error: Username required";
            return;
        }

        adminStatus = $"Unbanning {adminUser}...";

        try
        {
            var res = await AdminUnban(adminUser);
            adminStatus = res ? $"‚úì Unbanned {adminUser}" : $"‚úó Failed to unban {adminUser}";
            if (res) await RefreshPlayerList();
        }
        catch (Exception e)
        {
            adminStatus = "‚úó Error: " + e.Message;
        }
    }

    public async void CallAdminGetAll()
    {
        adminStatus = "Loading all players...";

        try
        {
            await RefreshPlayerList();
        }
        catch (Exception e)
        {
            adminStatus = "‚úó Error: " + e.Message;
        }
    }

    private async Task RefreshPlayerList()
    {
        var res = await AdminGetAllPlayers();

        if (res.success)
        {
            cachedList = res;
            int bannedCount = 0;

            foreach (var p in res.players)
            {
                if (p.banned == "true")
                    bannedCount++;
            }

            adminStatus = $"‚úì Loaded {res.players.Length} players ({bannedCount} banned)";
        }
        else
        {
            adminStatus = "‚úó Failed to load players";
        }
    }
}

Apps Script for Leaderboard

Leaderboard API - Code.gs 
function doGet(e) {
const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1");

const lastRow = sheet.getLastRow();
let leaderboard = [];

if (lastRow > 1) {
  const data = sheet.getRange(2, 1, lastRow - 1, 3).getValues();
  leaderboard = data.map(row => ({
    name: row[0],
    score: row[1],
    timestamp: row[2]
  }));
}

// Sort by score descending
leaderboard.sort((a, b) => b.score - a.score);

return ContentService
  .createTextOutput(JSON.stringify(leaderboard))
  .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1");

const body = JSON.parse(e.postData.contents);
const name = body.name;
const score = Number(body.score);

const lastRow = sheet.getLastRow();
let updated = false;

// If there are existing players
if (lastRow > 1) {
  const data = sheet.getRange(2, 1, lastRow - 1, 3).getValues();

  for (let i = 0; i < data.length; i++) {
    if (data[i][0] === name) {
      // ALWAYS overwrite existing player's score
      sheet.getRange(i + 2, 2).setValue(score);
      sheet.getRange(i + 2, 3).setValue(new Date());
      updated = true;
      break;
    }
  }
}

// If no match, add new player
if (!updated) {
  sheet.appendRow([name, score, new Date()]);
}

return ContentService
  .createTextOutput(JSON.stringify({ status: "success" }))
  .setMimeType(ContentService.MimeType.JSON);
}

Unity Leaderboard API

Create a reusable API class for leaderboard operations:

LeaderboardAPI.cs 
using System;
using System.Collections.Generic;
using System.Text;
using System.Threading.Tasks;
using UnityEngine;
using UnityEngine.Networking;

public static class LeaderboardAPI
{
  // Submit Score
  public static async Task<bool> SubmitScoreAsync(string baseUrl, string playerName, int score)
  {
      ScoreRequest data = new ScoreRequest(playerName, score);
      string json = JsonUtility.ToJson(data);
      
      UnityWebRequest req = new UnityWebRequest(baseUrl, "POST");
      req.uploadHandler = new UploadHandlerRaw(Encoding.UTF8.GetBytes(json));
      req.downloadHandler = new DownloadHandlerBuffer();
      req.SetRequestHeader("Content-Type", "application/json");
      req.SetRequestHeader("Accept", "application/json");
      req.SetRequestHeader("Access-Control-Allow-Origin", "*");

      var asyncOp = req.SendWebRequest();
      while (!asyncOp.isDone)
          await Task.Yield();

      if (req.result == UnityWebRequest.Result.Success)
      {
          Debug.Log("Score submitted: " + req.downloadHandler.text);
          req.Dispose();
          return true;
      }
      else
      {
          Debug.LogError("Error submitting score: " + req.error);
          req.Dispose();
          return false;
      }
  }

  // Get Leaderboard
  public static async Task<List<LeaderboardEntry>> GetLeaderboardAsync(string baseUrl)
  {
      UnityWebRequest req = UnityWebRequest.Get(baseUrl);
      var asyncOp = req.SendWebRequest();
      
      while (!asyncOp.isDone)
          await Task.Yield();

      if (req.result != UnityWebRequest.Result.Success)
      {
          Debug.LogError("Error fetching leaderboard: " + req.error);
          req.Dispose();
          return null;
      }

      string json = req.downloadHandler.text;
      req.Dispose();
      return ConvertJson(json);
  }

  // JSON Parsing
  private static List<LeaderboardEntry> ConvertJson(string json)
  {
      Wrapper wrapper = JsonUtility.FromJson<Wrapper>("{\"items\":" + json + "}");
      return wrapper.items;
  }

  // Data Structures
  [Serializable]
  private class Wrapper { public List<LeaderboardEntry> items; }

  [Serializable]
  public class LeaderboardEntry
  {
      public string name;
      public int score;
      public string timestamp;
  }

  [Serializable]
  private class ScoreRequest
  {
      public string name;
      public int score;
      
      public ScoreRequest(string n, int s)
      {
          name = n;
          score = s;
      }
  }
}

Unity Leaderboard Manager

Create a manager to handle leaderboard UI and interactions:

LeaderboardManager.cs 
using System.Collections.Generic;
using UnityEngine;

public class LeaderboardManager : MonoBehaviour
{
  public static string leaderboardURL = "YOUR_LEADERBOARD_URL_HERE";
  
  public Vector2 scrollPos;
  public bool isLoading = false;
  public string status = "";
  public List<LeaderboardAPI.LeaderboardEntry> leaderboard = new List<LeaderboardAPI.LeaderboardEntry>();

  public async void SubmitScore()
  {
      if (isLoading) return;
      
      isLoading = true;
      status = "Submitting...";

      string playerName = "Player_" + Random.Range(1000, 9999);
      int randomScore = Random.Range(0, 500);

      bool success = await LeaderboardAPI.SubmitScoreAsync(leaderboardURL, playerName, randomScore);

      status = success 
          ? $"Score submitted! ({playerName}: {randomScore})" 
          : "Failed to submit score.";
      
      isLoading = false;

      // Auto-refresh after submit
      if (success)
      {
          await System.Threading.Tasks.Task.Delay(500);
          FetchLeaderboard();
      }
  }

  public async void FetchLeaderboard()
  {
      if (isLoading) return;
      
      isLoading = true;
      status = "Loading leaderboard...";

      var result = await LeaderboardAPI.GetLeaderboardAsync(leaderboardURL);

      if (result != null)
      {
          leaderboard = result;
          status = $"Loaded! ({leaderboard.Count} entries)";
      }
      else
      {
          status = "Failed to load.";
      }

      isLoading = false;
  }
}

Best Practices

üîê Security First

Never store passwords in plain text. Use hashing algorithms and consider OAuth providers for production games.

üíæ Cache User Sessions

Store login tokens locally to avoid repeated authentication. Implement session expiry for security.

üö¶ Rate Limiting

Add cooldowns on login attempts to prevent brute-force attacks. Apps Script has built-in quotas.

üìä Data Validation

Always validate inputs on both client and server. Check username length, password strength, etc.

Testing Your System

Regardless of whether you're testing Remote Config, Player Authentication, or the Leaderboard, the testing workflow remains mostly the same. Use the following universal checklist:

  1. Deploy your Google Apps Script backend and copy the deployment URL.
  2. Paste the URL into your Unity script (RemoteConfigManager, PlayerManager, LeaderboardManager, etc.).
  3. Create a simple Unity test scene with UI buttons or input fields.
  4. Trigger your manager method (fetch, submit, update, login, etc.) using the button.
  5. Run the scene and check values in the Unity Console or Inspector.
  6. Modify values in Google Sheets and repeat the request to confirm updates.
  7. Verify all requests and responses in Chrome DevTools ‚Üí Network tab.
  8. Ensure error cases work correctly (empty fields, wrong input, missing data).
üí° Pro Tip: Always test with both valid and invalid inputs to ensure your Apps Script handles errors safely and returns clean JSON.

Troubleshooting

‚ùå "JSON Parse Error"

Solution: Check that column names in Google Sheets match your C# class fields exactly (case-sensitive). Enable Debug.Log to see raw response.

‚ùå POST Request Fails

Solution: Ensure you deployed as "Anyone" access. Check that you're using doPost() not doGet() for player operations. Verify Content-Type headers.

‚ùå "User already exists" on First Registration

Solution: Make sure your Players sheet has a header row. The script skips row 0 (headers) when checking usernames.

‚ùå WebGL Build Errors

Solution: Add the "Accept: application/json" header. WebGL has CORS restrictions that require explicit headers.
